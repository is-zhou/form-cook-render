InputTag 输入框 ​

基础用法 ​

按 Enter 回车键添加输入内容为标签

ts
{
  id: "",
  componentName: "InputTag",
  componentType: "form",
  formItemAttrs: {
    field: "input",
    label: "",
  },
  attrs: {
    placeholder: "请输入",
  },
}

自定义触发器 ​

ts
{
  id: "",
  componentName: "InputTag",
  componentType: "form",
  formItemAttrs: {
    field: "input",
    label: "",
  },
  attrs: {
    placeholder: "请输入",
    trigger:"Enter" // "Enter"|"Space"
  },
}

最大标签数 ​

ts
{
  id: "",
  componentName: "InputTag",
  componentType: "form",
  formItemAttrs: {
    field: "input",
    label: "",
  },
  attrs: {
    placeholder: "请输入",
    max:3
  },
}

折叠标签 ​

ts
{
  id: "",
  componentName: "InputTag",
  componentType: "form",
  formItemAttrs: {
    field: "input",
    label: "",
  },
  attrs: {
    placeholder: "请输入",
    'collapse-tags':true, 
    'collapse-tags-tooltip'：true,
    'max-collapse-tags':3,
  },
}

可拖放 ​

ts
{
  id: "",
  componentName: "InputTag",
  componentType: "form",
  formItemAttrs: {
    field: "input",
    label: "",
  },
  attrs: {
    placeholder: "请输入",
    draggable: true, 
  }, 
  defaultValue: ["tag1", "tag2", "tag3"], 
}

其他 ​

ts
{
  id: "",
  componentName: "InputTag",
  componentType: "form",
  formItemAttrs: {
    field: "input",
    label: "",
  },
  attrs: {
    placeholder: "请输入",
    disabled: true, //禁用
    clearable: true,//可清空
    delimiter:',', //分隔符
    size:"large",//尺寸
  },
}
